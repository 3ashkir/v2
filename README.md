function parseAndAdd(){
const raw=document.getElementById('admOrders').value.trim();
if(!raw){toast('Paste some orders first');return}
const dept=document.getElementById('admDept').value;
const status=document.getElementById('admStatus').value;
const lines=raw.split('\n').filter(l=>l.trim());
const log=document.getElementById('admLog');
log.innerHTML='';
let added=0,failed=0;

if(!DEL[dept])DEL[dept]=[];

lines.forEach(line=>{
line=line.trim();
// Try pipe-separated: ID | Name | Qty | UOM
let parts=line.split('|').map(s=>s.trim());
if(parts.length<3){
// Try tab-separated
parts=line.split('\t').map(s=>s.trim());
}
if(parts.length>=3){
const id=parts[0].replace(/[^0-9]/g,'');
const name=parts[1];
const qty=parseInt(parts[2])||1;
const uom=parts[3]||'Each';
if(id&&name){
// Check for duplicate
const exists=DEL[dept].find(x=>x.i===id&&x.q===qty);
if(exists){
log.innerHTML+=`<div class="er">‚ö† Duplicate: ${name} (#${id}) - skipped</div>`;
return;
}
DEL[dept].push({i:id,n:name,q:qty,u:uom,s:status});
log.innerHTML+=`<div class="ok">‚úì Added: ${name} (${qty} ${uom}) ‚Üí ${dept}</div>`;
added++;
return;
}
}
// Couldn't parse
log.innerHTML+=`<div class="er">‚úó Can't parse: "${line.substring(0,50)}"</div>`;
failed++;
});

if(added>0){
saveOrders();
addLog('Added '+added+' orders to '+dept);
toast('‚úì Added '+added+' order'+(added>1?'s':''));
updateSummary();
}else{
toast('No orders could be parsed');
}
}
